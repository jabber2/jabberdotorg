is to define it as a function...

#if defined I_HAVE_A_LAME_COMPILER

  terror tmp;
  tmp.code=405;
  strcpy(tmp.msg,"Not Allowed");
  return tmp;
}
#define TERROR_NOTALLOWED TERROR_NOTALLOWED_func()
#else
#define TERROR_NOTALLOWED (terror) { 405, "Not Allowed" }
#endif
