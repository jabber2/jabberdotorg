I don't know, it's a standard configuration:

    <service id="msn.charente.de">
        <host>conference.msn.charente.de</host>

        <msntrans xmlns="jabber:config:msntrans">
            <instructions>Geben Sie bitte Ihren MSN-Messenger account 
(z.B. user1 at hotmail.com und Passwort ein. Eine Hotmail-Adresse ist z.B. 
automatisch ein MSN-Messenger-Account.</instructions>
            <vCard>
                <FN>MSN-Gateway</FN>
                <DESC>Mit diesem Uebergang ist die Kommunikation mit 
MSN-Messenger-Benutzern moeglich.</DESC>
                <URL>http://charente.de/jabber/</URL>
            </vCard>

            <conference id="conference.msn.charente.de">
                <invite>Mehr als ein Nutzer hat diesen Chat betreten. In 
diesen Chatraum eintreten um in den Gruppenchat-Modus zu wechseln.</invite>
                <notice>
                    <join> ist eingetreten</join>
                    <leave> ist gegangen</leave>
                </notice>
            </conference>
            <!-- The default server(s) is usually ok
            <servers>
                <ip>64.4.13.56</ip>
                <ip>64.4.13.57</ip>
            </servers> -->

            <headlines/>  <!-- Enable Hotmail inbox notification -->

        </msntrans>
        <load>
            <msntrans>./msn-transport/src/msntrans.so</msntrans>
        </load>
    </service>
