> Could you be more specific on this issue? The following is an example of
> <xdb> in jabber.xml:
yep, you can have something like that :

  <xdb id="xdb_file">
    <host/>
    <ns>jabber:xdb:nslist</ns>
    ...
    <load>
      <xdb_file>./xdb_file/xdb_file.so</xdb_file>
    </load>
    <xdb_file xmlns="jabber:config:xdb_file">
      <spool><jabberd:cmdline flag='s'>./spool</jabberd:cmdline></spool>
    </xdb_file>
  </xdb>
  
  <xdb id="xdb_sql">
    <host/>
    <ns>jabber:iq:roster</ns>
    ...
    <load>
      <xdb_sql>./xdb_sql/xdb_sql.so</xdb_sql>
    </load>
    <jabberd:include>./xdb_sql/xdb_sql.xml</jabberd:include>
  </xdb>
