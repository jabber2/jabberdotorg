Hello everyone!

I'm running a jabberd-1.4.2 on my Win2000, and have meet the problem:
There is an old and bugus third-party ICQ Server, supporting icqv5 and icqv3
only (that's why I cannot use icqv7-t or aim). It is periodically ignoring
ACK packets (in my case those ones were acknowledging SRV_USER_META) going
from transport and killing the session (sending SRV_GO_AWAY). After that
transport also kills connection and reports a error.

Is there any possible method to make transport reconnect after such error is
received? I tried to modify `session.c' such way (posted below). Now
connection is restored after described failure, but after reconnect
transport is totally ignoring all incoming ICQ packets, and does not send
any ICQ packets at all. Debug log from server is attached (transport is
running in separate process).

-------- Here is modified
session.c ----------------------------------------------

void it_session_error(session s, terror e)
{
    jpq jq;

    if (s->exit_flag == 1)
        return;

    if (s->server)
    {
        if (s->connected)
            it_packet_text_code(s,"B_USER_DISCONNECTED");
        s->server->cb_arg = NULL;
        mio_close(s->server);
    }

    if (s->peer_accept)
       mio_close(s->peer_accept);

    s->rsend = -1;
    s->type = stype_normal;
    s->addr.sin_family = AF_INET;
    s->addr.sin_addr.s_addr = 0x0100007FUL;

    s->sid = (unsigned long) random();
    s->seq1 = (unsigned short)(random() & 0x7FFF);
    s->seq2 = 1;

    log_debug (ZONE, "_av: Dropping all undelivered packets");
    while (s->pq_head != NULL)
        it_server_acked_packet (s, s->pq_head->seq1);

    if (!it_server_connect (s) && (s->ti->tcp == TCP_FLAG_NOTCP ||
!it_peer_listen (s)))
    {
        it_packet_login (s);
        log_debug(ZONE, "_av: reinitializing session");
        return;
    }

    log_debug(ZONE,"Killing session[%s], '%s'",jid_full(s->id),e.msg);

    s->exit_flag = 1;

    if (s->type != stype_normal)
        it_session_regerr(s,e);
    else
        it_session_unavail(s,e.msg);

    mtq_send(s->q,s->p,it_session_exit,(void *) s);
}
