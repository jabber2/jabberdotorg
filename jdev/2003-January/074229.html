You can configure the server and charmap the transport is using in its 
configuration file:

<config xmlns="jabber:config:yahoo">
    <vCard>
       <!-- your transport's vCard -->
    </vCard>
    <server>cs.yahoo.co.jp</server>
    <port>5050</port>
    <charmap>SHIFT-JIS</charmap>
</config>
